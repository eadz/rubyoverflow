    def get_post(parameters={})
      if @post_url
        hash,url = request(@post_url, parameters)
        if @post_url.include? 'question'
          Questions.new hash, url 
        elsif @post_url.include? 'answer'
          Answers.new hash, url
        end
      else
        nil
      end
    end

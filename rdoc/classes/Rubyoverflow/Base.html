    def find_parse_querystring(rawurl)
      url, querystring = rawurl.split('?')
      queryHash = {}
      if querystring
        querystring.split('&').each {|pair|
          key,value=pair.split('=')
          queryHash[key]=value 
        }
        queryHash
      end
      url.sub!(Base.client.host_path,'')
      return url,queryHash
    end
